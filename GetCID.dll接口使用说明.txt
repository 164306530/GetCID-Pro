根据安装ID在线获取确认ID模块.
.net4.6库,使用说明：

vb.net:

                            Dim func As New GetCID.GetCID
                            Dim flags As Integer = func.GetConfirmationID(InstalltionID, ConfirmationID)
                            Do
                                If flags <> 0 Then
                                    Exit Do
                                End If
                                Application.DoEvents()
                            Loop
                            Debug.Print(ConfirmationID)


c#:

        private static T InlineAssignHelper<T>(ref T target, T value)
        {
            target = value;
            return value;
        }
        private void Button1_Click(object sender, EventArgs e)
        {
            int nFlag = 0;
            string ConfirmationID = "";
            GetCID.GetCID func = new GetCID.GetCID();
            System.Threading.Thread Thread = new System.Threading.Thread(() => InlineAssignHelper(ref nFlag, func.GetConfirmationID(textBox1.Text,ref ConfirmationID)));
            Thread.Start();
            do
            {
                if (nFlag != 0)
                {
                    break;
                }
                Application.DoEvents();
            } while (true);
            Thread.Abort();
            textBox2.Text = ConfirmationID;
        }
